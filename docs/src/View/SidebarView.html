 93    def _create_button(self) -> customtkinter.CTkButton:
 94        '''
 95        Creates the button that is used to switch from one task list view
 96        to another. Double clicking on the button opens an entry for
 97        modifying the task list's name.
 98        '''
 99        button = customtkinter.CTkButton(
100                master=self,
101                fg_color='transparent',
102                text_color=self.highlight_color,
103                text_color_disabled=('black', 'white'),
104                corner_radius=0,
105                text=' ')   # allows button to initialize before entry
106
107        # partial is used to pass objects from the current scope as arguments
108        button.bind('<Button-1>', partial(self.master.select_list, button))
109
110        # display entry form for renaming the task list when double clicked
111        button.bind('<Double-Button-1>', self.display_entry)       
112
113        # highlight the background when moused over
114        button.bind("<Enter>", self._highlight_button)
115
116        # return to normal when mouse leaves
117        button.bind("<Leave>", self._reset_button_style)
118
119        button.grid(row=0, column=0)
120        return button

In the text entry state, a text entry appears over top of the button, hiding it from view. Upon appearing, the text entry will contain the text from the buttons label. The entry will be shown in two distinct case.

  1. The button is double clicked
  2. The widget is created with out a title
